
Phantom server install

## install node and npm
sudo apt-get install nodejs npm phantomjs fontconfig libfreetype6 libfreetype6-dev git nginx

## enable multiverse
sudo vi /etc/apt/sources.list
## now search for first two instance of deb multiverse (ignore deb-src) and uncomment them

## install the ms fonts
sudo apt-get install ttf-mscorefonts-installer

## disable multiverse
sudo vi /etc/apt/sources.list
## now comment out the two lines you previous uncommented (deb multiverse)

## load fontconfig and reload the cache
sudo fc-cache -fv

## let's get the node plugins we need
sudo mkdir /phantom
cd /phantom
unzip the contents of SGX phantom dir here

## install forever (need to install after every unpack)
sudo npm install forever -g

## fix ubuntu pathing thing
sudo ln -s /usr/bin/nodejs /usr/bin/node

## kick off transform server 
sudo forever start server.js

## now let's set up nginx
sudo mv nginx.conf /etc/nginx/sites-enabled/default
sudo nginx -s reload
